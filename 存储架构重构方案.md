# 存储架构重构方案

## 一、当前问题分析

### 1.1 存储位置混乱

| 问题 | 描述 |
|------|------|
| 配置分散 | 用户配置在 uTools DB，学习数据在 IndexedDB，自定义词库又在 uTools DB |
| 职责不清 | uTools DB 既是配置存储，又是词库数据存储 |
| 耦合度高 | 存储层与业务逻辑紧耦合，难以测试和维护 |

### 1.2 数据冗余

| 表 | 问题 |
|----|------|
| `learningRecords` | 会话统计，可从 `wordRecords` 聚合 |
| `dailyRecords` | 每日统计，可从 `wordRecords` 聚合 |
| `dictProgress` | 词库进度，可从 `wordProgress` 聚合 |

### 1.3 查询效率低

| 问题位置 | 描述 |
|----------|------|
| `typingStates` | 全量 `toArray()` 后再 filter |
| `wordProgress` | 多处 `where().toArray()` 全量加载 |
| `wordRecords` | 统计页面全量加载所有记录 |

### 1.4 临时状态持久化

| 表 | 问题 |
|----|------|
| `typingStates` | 重复学习的临时状态，不应持久化到 IndexedDB |

### 1.5 代码重复

| 重复代码 | 位置 |
|----------|------|
| `useDailyRecord` vs `DailyRecordService` | 功能完全重复 |
| `useWordProgress` vs `WordProgressService` | 功能完全重复 |
| 多处 `where().toArray()` | 相同查询模式重复出现 |

```typescript
export interface IStorageAdapter {
  get<T>(key: string): Promise<T | null>
  set<T>(key: string, value: T): Promise<void>
  delete(key: string): Promise<void>
  clear(): Promise<void>
}

/**
 * 数据库接口 - IndexedDB 专用
 */
export interface IDatabaseAdapter {
  getTable<T>(tableName: string): ITableAdapter<T>
}

export interface ITableAdapter<T> {
  get(id: number): Promise<T | undefined>
  add(record: Omit<T, 'id'>): Promise<number>
  update(id: number, changes: Partial<T>): Promise<void>
  delete(id: number): Promise<void>
  where(index: string): IWhereClause<T>
  toArray(): Promise<T[]>
  count(): Promise<number>
}

export interface IWhereClause<T> {
  equals(value: unknown): IWhereClause<T>
  anyOf(values: unknown[]): IWhereClause<T>
  between(lower: unknown, upper: unknown): IWhereClause<T>
  first(): Promise<T | undefined>
  toArray(): Promise<T[]>
}
```

#### 3.1.2 创建 Repository 层

```typescript
// src/storage/repositories/WordRecordRepository.ts

export class WordRecordRepository {
  constructor(private db: IDatabaseAdapter) {}

  /**
   * 添加单词记录
   */
  async add(record: Omit<IWordRecord, 'id'>): Promise<number> {
    return this.db.getTable<IWordRecord>('wordRecords').add(record)
  }

  /**
   * 获取词库在某时间范围内的记录
   */
  async getRecordsInRange(
    dict: string,
    startTime: number,
    endTime: number
  ): Promise<IWordRecord[]> {
    return this.db.getTable<IWordRecord>('wordRecords')
      .where('[dict+timeStamp]')
      .between([dict, startTime], [dict, endTime])
      .toArray()
  }

  /**
   * 获取某天的学习单词列表（用于重复学习）
   */
  async getTodayLearnedWords(dict: string, date: string): Promise<string[]> {
    const { startOfDay, endOfDay } = getDayRange(date)
    const records = await this.getRecordsInRange(dict, startOfDay, endOfDay)
    return [...new Set(records.map(r => r.word))]
  }

  /**
   * 统计某天的学习数据
   */
  async getDayStats(dict: string, date: string): Promise<DayStats> {
    const { startOfDay, endOfDay } = getDayRange(date)
    const records = await this.getRecordsInRange(dict, startOfDay, endOfDay)
    
    return {
      totalWords: new Set(records.map(r => r.word)).size,
      totalInputs: records.length,
      wrongCount: records.reduce((sum, r) => sum + r.wrongCount, 0),
    }
  }
}
```

```typescript
// src/storage/repositories/WordProgressRepository.ts

export class WordProgressRepository {
  constructor(private db: IDatabaseAdapter) {}

  /**
   * 获取单词进度
   */
  async getProgress(dict: string, word: string): Promise<IWordProgress | undefined> {
    return this.db.getTable<IWordProgress>('wordProgress')
      .where('[dict+word]')
      .equals([dict, word])
      .first()
  }

  /**
   * 批量获取单词进度
   */
  async getProgressBatch(dict: string, words: string[]): Promise<Map<string, IWordProgress>> {
    const keys = words.map(w => [dict, w])
    const records = await this.db.getTable<IWordProgress>('wordProgress')
      .where('[dict+word]')
      .anyOf(keys)
      .toArray()
    
    return new Map(records.map(r => [r.word, r]))
  }

  /**
   * 获取到期复习的单词
   */
  async getDueWords(dict: string, now: number): Promise<IWordProgress[]> {
    return this.db.getTable<IWordProgress>('wordProgress')
      .where('nextReviewTime')
      .belowOrEqual(now)
      .filter(p => p.dict === dict)
      .toArray()
  }

  /**
   * 获取新单词（未学习过的）
   */
  async getNewWords(dict: string, limit: number): Promise<IWordProgress[]> {
    return this.db.getTable<IWordProgress>('wordProgress')
      .where('[dict+masteryLevel]')
      .equals([dict, 0])
      .limit(limit)
      .toArray()
  }

  /**
   * 更新单词进度
   */
  async updateProgress(id: number, changes: Partial<IWordProgress>): Promise<void> {
    return this.db.getTable<IWordProgress>('wordProgress').update(id, changes)
  }

  /**
   * 获取词库统计
   */
  async getDictStats(dict: string): Promise<DictStats> {
    const all = await this.db.getTable<IWordProgress>('wordProgress')
      .where('dict')
      .equals(dict)
      .toArray()
    
    return {
      total: all.length,
      learning: all.filter(p => p.masteryLevel > 0 && p.masteryLevel < 5).length,
      mastered: all.filter(p => p.masteryLevel >= 5).length,
      new: all.filter(p => p.masteryLevel === 0).length,
    }
  }
}
```

### 3.2 服务层重构

#### 3.2.1 创建统一的 LearningService

```typescript
// src/services/LearningService.ts

export class LearningService {
  private wordRecordRepo: WordRecordRepository
  private wordProgressRepo: WordProgressRepository
  private configRepo: ConfigRepository

  constructor(
    wordRecordRepo: WordRecordRepository,
    wordProgressRepo: WordProgressRepository,
    configRepo: ConfigRepository
  ) {
    this.wordRecordRepo = wordRecordRepo
    this.wordProgressRepo = wordProgressRepo
    this.configRepo = configRepo
  }

  /**
   * 开始学习会话
   */
  async startSession(dict: string): Promise<LearningSession> {
    const config = await this.configRepo.getDailyLimitConfig()
    const dueWords = await this.wordProgressRepo.getDueWords(dict, Date.now())
    const newWords = await this.wordProgressRepo.getNewWords(dict, config.dailyLimit)
    
    return {
      dict,
      startTime: Date.now(),
      dueWords,
      newWords,
      learnedToday: 0, // 从缓存或实时计算
    }
  }

  /**
   * 记录单词学习
   */
  async recordWord(
    word: string,
    dict: string,
    result: { correct: boolean; wrongCount: number; timing: number[] }
  ): Promise<void> {
    // 1. 保存单词记录
    await this.wordRecordRepo.add({
      word,
      dict,
      timeStamp: Date.now(),
      wrongCount: result.wrongCount,
      timing: result.timing,
      // ... 其他字段
    })

    // 2. 更新单词进度
    const progress = await this.wordProgressRepo.getProgress(dict, word)
    if (progress) {
      const newProgress = calculateSRS(progress, result.correct)
      await this.wordProgressRepo.updateProgress(progress.id!, newProgress)
    }
  }

  /**
   * 获取今日学习统计
   */
  async getTodayStats(dict: string): Promise<TodayStats> {
    const today = getTodayString()
    return this.wordRecordRepo.getDayStats(dict, today)
  }

  /**
   * 获取重复学习单词列表
   */
  async getRepeatLearningWords(dict: string): Promise<string[]> {
    const today = getTodayString()
    return this.wordRecordRepo.getTodayLearnedWords(dict, today)
  }
}
```

```typescript
export async function migrateToV2(): Promise<void> {
  const oldVersion = await db.version()
  
  if (oldVersion < 6) {
    // 迁移 learningRecords 数据到 wordRecords（如果需要）

      wordProgress: '++id, [dict+word], [dict+masteryLevel], nextReviewTime, lastReviewTime',
```
